class factorialnumbers
{

	public static void main(String[] args) 
	{	

		System.out.println("factorial numbers");                                                                                  
			int k;                                                                                                                                                  
			int fact=1;                                                                                                                                             
			int n=5;                                                                                                                                                
			for(k=1;k<=n;k++)                                                                                                                   
			{                                                                                                                                                       
				fact=fact*k;                                                                                                                    
			}   
			System.out.println(fact); 

	}
}
